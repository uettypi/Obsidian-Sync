
### 第一阶段：赛题基础信息与分析切入点
将比赛附件解压后，我们会得到一个被剥离了符号表（stripped）的 PIE ELF 执行文件。尝试化运行该文件时能够发现：若提供错误的 flag，程序仅返回 `no` 字符串；只有当输入完全校验通过时，才会打印出 `ok`。此外，经过初步分析发现该二进制体内藏有多重对抗动态调试和追踪的保护措施，具体涵盖了 `ptrace`、`/proc/self/status`、`/proc/<ppid>/comm` 以及 `clock_gettime` 等系统调用特性。

通过对 PLT 表项、接收输入的节点及各类输出字符串对应的指令块（gadget）进行逆向溯源，可以清晰判定用户输入的读取位置大致落在 `0x15a7f` 附近，且执行流会根据校验结果的不同，分别引导向打印 `no` 或是 `ok` 的终端汇编分支。再结合本题的命名“北极星混淆”，可进而推侧该程序采用了类似于 Polaris 的间接跳转保护机制与控制流平坦化方案，导致其真正负责数据加密的核心运算逻辑完全潜藏在冗杂的状态机转换以及毫无意义的垃圾指令之中。

### 第二阶段：剥离冗余混淆与核心逻辑重建
首要步骤是规避内置的反调试机制，进而解析出用于字符串还原的辅助功能（helper 函数）。经过静态观察与调试配合，该 helper 在本质上执行的是一套基于特定 key 的周期性异或（XOR）解密运算。依靠这一底层逻辑，我们能顺畅地将其用于反调试环境监测的关键特征码悉数提取并还原，这包含了 `/proc/self/status`、`TracerPid`、`gdb`、`strace`、`ltrace`、`frida` 以及 `radare` 等相关词汇。

接下来，我们将视线转移至程序运行期查表动态构建流程中进行溯源。可以观察到程序刚启动伊始，会依托一个固定的 `seed` 值来动态生成一整套复杂的白盒密码映射表：其中，`T0` 代表最终用于对比匹配的 48 字节长度的密文；`T1` 是用于 10 轮变换计算的 16×256 字节矩阵表；`T2` 扮演了 9 轮运算所需的 16×4×256 字节混合参数表的角色；而 `T3` 则充当最终结果导出前的统一输出编码转化表。

在这个基础上，如果对内存中访问 `T1`、`T2` 乃至 `T3` 此类数据表的时序进行访问路径的插桩记录，就能彻底印证其加密框架是相当标准的白盒 AES 体制：前置的 9 轮均依赖 `T1` 完成迭代，并配合 `ShiftRows` 轮转与 `MixColumns` 列混淆实施查表并执行异或指令；紧接着在收尾的最后一轮里，先凭借 `T1` 查表再照着 `ShiftRows` 次序完成交织，最终将结果投喂至 `T3` 表内进行深度编码输出。上部分所提及的 `T0`，也就是对应此程序在其执行终点处进行单字节逐一比对的最终校验标尺。

正因如此，我们在此毫无必要调用大规模算力进行暴力求举。思路应当转为在脚本环境重现出那组别无二致的映射表，接着循着白盒 AES 的底层框架进行倒推操作：具体做法为优先反向解析 `T3` 以及最尾部一轮的 `T1`；其后针对余下的 9 轮步骤，逐列应用相遇中间人攻击算法（meet-in-the-middle）实施纵列溯源，以求得 4 字节 `MixColumns` 的反向映射；等到全部十六进制明文矩阵还原浮出水面以后，再去斩除掉块数据残余的 `PKCS#7` padding 填充即可。

### 第三阶段：编写自动化 EXP 与有效性复现
利用下方编写的 `exp.py` 自动化开发脚本，可以将 ELF 二进制内部存在的加密表溯源数据完整读取进脚本生态中，自主模拟重演并映射出一套运行时查表体系，再以此实现 `T0` 数值的逆向还原获取对应明文串。更为严谨的是，该代码末端囊括有附加了 3 秒时限控制的系统调用项，使得脚本可以带着反推出的内容回调真实程序进行通过性测试。代码回放结果请阅下方。


```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""ISCC 北极星混淆 通用 EXP：自动还原运行时白盒表并反解 flag。"""
import os, struct, math, subprocess, sys

try:
    from elftools.elf.elffile import ELFFile
except ImportError:
    print('[!] need pyelftools: pip install pyelftools', file=sys.stderr)
    raise

BIN = './traditional_re'
MASK = (1 << 64) - 1
SHIFT_COLS = [[0, 5, 10, 15], [4, 9, 14, 3], [8, 13, 2, 7], [12, 1, 6, 11]]

def load_elf(path):
    f = open(path, 'rb')
    elf = ELFFile(f)
    f.seek(0)
    data = f.read()
    segs = [seg for seg in elf.iter_segments() if seg['p_type'] == 'PT_LOAD']
    def off(vaddr):
        for seg in segs:
            if seg['p_vaddr'] <= vaddr < seg['p_vaddr'] + seg['p_filesz']:
                return seg['p_offset'] + (vaddr - seg['p_vaddr'])
        raise ValueError(hex(vaddr))
    def rb(vaddr, n):
        o = off(vaddr)
        return data[o:o+n]
    def ptrs(vaddr, n):
        return struct.unpack('<' + 'Q' * n, rb(vaddr, 8*n))
    return rb, ptrs

rb, ptrs = load_elf(BIN)

def splitmix_step(state):
    state = (state + 0x9e3779b97f4a7c15) & MASK
    z = state
    z = ((z ^ (z >> 30)) * 0xbf58476d1ce4e5b9) & MASK
    z = ((z ^ (z >> 27)) * 0x94d049bb133111eb) & MASK
    return state, (z ^ (z >> 31)) & MASK

def hashseed(seed, n, flag_byte):
    x = (seed ^ ((n * 0xd1342543de82ef95) & MASK) ^ ((flag_byte * 0x6a09e667f3bcc909) & MASK)) & MASK
    acc = 0
    for _ in range(4):
        st = x ^ acc
        st, out = splitmix_step(st)
        x = st
        acc ^= out
    return acc

def coprime_step(n, seed):
    s = ((seed >> 17) | 1) % n or 1
    while math.gcd(s, n) != 1:
        s = (s + 2) % n or 1
    return s

def table_get(rows, n, seed, i):
    row = (i * coprime_step(n, seed) + ((seed >> 23) % n)) % n
    col = i // n
    return rb(rows[row] + col, 1)[0]

def gen_table(src_ptr_table, out_len, seed, rows_n):
    rows = ptrs(src_ptr_table, rows_n)
    s0 = hashseed(seed, out_len, 0)
    step = coprime_step(out_len, s0)
    base = (hashseed(0xa5a5a5a55a5a5a5a ^ s0, out_len, 0) >> 19) % out_len
    cur = s0
    out = bytearray(out_len)
    for i in range(out_len):
        dst = (i * step + base) % out_len
        cur, rnd = splitmix_step(cur)
        b = (rnd >> ((i & 7) * 8)) & 0xff
        b ^= ((dst * 0xa7 + i * 0x3d) & 0xff)
        b ^= table_get(rows, rows_n, seed, i)
        out[dst] = b
    return bytes(out)

T0 = gen_table(0x183bc0, 0x30,    0x61f29c0d87a453b1, 5)   # target ciphertext
T1 = gen_table(0x183bf0, 0xa000,  0xd3b5407c19ea8f26, 11)  # round byte tables
T2 = gen_table(0x183c50, 0x24000, 0x7a8ce153d604b92f, 13)  # mix tables
T3 = gen_table(0x183cc0, 0x1000,  0xc94e2f81a73d650b, 7)   # final output tables

INV_T1 = []
for r in range(10):
    inv_round = []
    for p in range(16):
        inv = [None] * 256
        for x in range(256):
            inv[T1[r*0x1000 + p*0x100 + x]] = x
        inv_round.append(inv)
    INV_T1.append(inv_round)

INV_T3 = []
for p in range(16):
    inv = [None] * 256
    for x in range(256):
        inv[T3[p*0x100 + x]] = x
    INV_T3.append(inv)

def invert_mix_column(next_state, round_id, col):
    target = tuple(next_state[4*col + i] for i in range(4))
    left = {}
    for a in range(256):
        for b in range(256):
            vec = []
            for row in range(4):
                outpos = 4*col + row
                base = round_id*0x4000 + outpos*0x400
                vec.append(T2[base + a] ^ T2[base + 0x100 + b])
            left[tuple(vec)] = (a, b)
    for c in range(256):
        for d in range(256):
            need = []
            for row in range(4):
                outpos = 4*col + row
                base = round_id*0x4000 + outpos*0x400
                need.append(T2[base + 0x200 + c] ^ T2[base + 0x300 + d] ^ target[row])
            if tuple(need) in left:
                return (*left[tuple(need)], c, d)
    raise RuntimeError('column inverse failed')

def dec_block(ct):
    order = sum(SHIFT_COLS, [])
    y = [0] * 16
    for outpos in range(16):
        y[order[outpos]] = INV_T3[outpos][ct[outpos]]
    state = [INV_T1[9][p][y[p]] for p in range(16)]
    for r in range(8, -1, -1):
        yy = [0] * 16
        for col, pos_list in enumerate(SHIFT_COLS):
            vals = invert_mix_column(state, r, col)
            for pos, val in zip(pos_list, vals):
                yy[pos] = val
        state = [INV_T1[r][p][yy[p]] for p in range(16)]
    return bytes(state)

plain = b''.join(dec_block(T0[i:i+16]) for i in range(0, len(T0), 16))
pad = plain[-1]
if 0 < pad <= 16 and plain.endswith(bytes([pad]) * pad):
    plain = plain[:-pad]

flag = plain.decode('ascii')
print(flag)

# 验证阶段带超时，防止异常卡死
try:
    r = subprocess.run([BIN], input=(flag + '\n').encode(), stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=3)
    print('[verify]', r.stdout.decode(errors='ignore').strip())
except subprocess.
TimeoutExpired:
    print('[verify] timeout')
```
