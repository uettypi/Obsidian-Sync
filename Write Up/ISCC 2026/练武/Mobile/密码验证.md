
## 解题过程

### 步骤 1 - 定位校验入口

**关键观察**

先看 Java/Kotlin 层，只能得到外层格式条件：总长度 36、前缀 `ISCC{`、后缀 `}`。真正的逻辑在 `NativeBridge.verifyFlag`，因此后续全部围绕 `libnativecrypto.so` 与 assets 分析。

### 步骤 2 - 还原 native 三段主约束

**关键观察**

JNI 会读取 `cipher1.bin`、`cipher2.bin`、`cipher3.bin`、`puzzle.bin`。先把 native 内置的三组 16 字节静态表按统一规则解码，再生成 3 个中间状态：第一组作为 TEA 密钥来源，第二组作为 GF(2^8) 线性变换矩阵，第三组作为前缀异或链与 LCG 种子来源。这样即可把 30 字节中段拆成 8 + 12 + 10 三段分别逆出。

### 步骤 3 - 恢复前 8 字节

**关键观察**

对 `cipher1.bin` 做 16 轮 TEA 逆运算，再与由中间状态导出的 8 字节掩码异或，得到中段前 8 字节。这个步骤不会依赖固定 flag，而是直接依赖当前附件里的 `cipher1.bin`。

### 步骤 4 - 恢复中间 12 字节

**关键观察**

把 GF(2^8) 线性变换矩阵求逆，对 `cipher2.bin` 按 3 行 4 列做逆变换，得到中段的 12 字节。native 中使用的乘法规则与 AES 常见有限域乘法一致，因此可以稳定脚本化恢复。

### 步骤 5 - 恢复最后 10 字节并做一致性校验

**关键观察**

用前面导出的 64 位种子跑 LCG 伪随机流，与 `cipher3.bin` 异或得到最后 10 字节。然后再用 `puzzle.bin` 的 4 行有限域校验、前 8 字节整体异或得到的字符映射、以及前 20 字节的 xor/sum 映射，验证三段是否属于同一份合法输入。

### 步骤 6 - 拼接最终 flag

**关键观察**

三段都恢复并通过附加约束后，直接拼回 `ISCC{<30-byte-body>}`。EXP 对当前 APK 重新读取 assets 并计算，因此同模板下只要附件内容变化，脚本会自动给出新的 flag。

## 最终验证

对当前附件运行通用 EXP 后，得到结果：`ISCC{jceuQHRS8%2ayJj^P:79HjwI$m*Bi2}`。

```python
#!/usr/bin/env python3
import sys
import zipfile
from functools import reduce


RAW_X = bytes.fromhex("5384792a17483deedb0c01b29fd0b566")
RAW_Y = bytes.fromhex("6e6d6d6d6d6e6d6d6d6d6e6d6d6d6d6e")
RAW_Z = bytes.fromhex("03f4a99a5e8b7c31224f40e5d66db7ca")
SBOX = bytes.fromhex(
    "637c777bf26b6fc53001672bfed7ab76ca82c97dfa5947f0add4a2af9ca472c0"
    "b7fd9326363ff7cc34a5e5f171d8311504c723c31896059a071280e2eb27b275"
    "09832c1a1b6e5aa0523bd6b329e32f8453d100ed20fcb15b6acbbe394a4c58cf"
    "d0efaafb434d338545f9027f503c9fa851a3408f929d38f5bcb6da2110fff3d2"
    "cd0c13ec5f974417c4a77e3d645d197360814fdc222a908846eeb814de5e0bdb"
    "e0323a0a4906245cc2d3ac629195e479e7c8376d8dd54ea96c56f4ea657aae08"
    "ba78252e1ca6b4c6e8dd741f4bbd8b8a703eb5664803f60e613557b986c11d9e"
    "e1f8981169d98e949b1e87e9ce5528df8ca1890dbfe6426841992d0fb054bb16"
)
CHARSET = b"0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz{}_-!@#$%^&*()[]<>?.,:;+/\\"


def gf_mul(a: int, b: int) -> int:
    out = 0
    while b:
        if b & 1:
            out ^= a
        hi = a & 0x80
        a = (a << 1) & 0xFF
        if hi:
            a ^= 0x1B
        b >>= 1
    return out


def tea_decrypt_block(block: bytes, key: bytes) -> bytes:
    v0 = int.from_bytes(block[:4], "big")
    v1 = int.from_bytes(block[4:], "big")
    words = [int.from_bytes(key[i : i + 4], "big") for i in range(0, 16, 4)]
    delta = 0x9E3779B9
    total = (delta * 16) & 0xFFFFFFFF

    for _ in range(16):
        v1 = (v1 - (((v0 << 4) + words[2]) ^ (v0 + total) ^ ((v0 >> 5) + words[3]))) & 0xFFFFFFFF
        v0 = (v0 - (((v1 << 4) + words[0]) ^ (v1 + total) ^ ((v1 >> 5) + words[1]))) & 0xFFFFFFFF
        total = (total - delta) & 0xFFFFFFFF

    return v0.to_bytes(4, "big") + v1.to_bytes(4, "big")


def lcg_stream(seed: int, length: int) -> bytes:
    state = seed
    out = bytearray()
    for _ in range(length):
        state = (state * 0x5851F42D4C957F2D + 0x14057B7EF767814F) & 0xFFFFFFFFFFFFFFFF
        out.append((state >> 24) & 0xFF)
    return bytes(out)


def decode_static_table(raw: bytes) -> bytes:
    return bytes((((byte - 0x13) & 0xFF) ^ 0x5A) for byte in raw)


def build_xyz() -> tuple[bytes, bytes, bytes]:
    t1 = decode_static_table(RAW_X)
    t2 = decode_static_table(RAW_Y)
    t3 = decode_static_table(RAW_Z)

    x = bytes((a ^ b) & 0xFF for a, b in zip(t1, t3))
    x = x[3:] + x[:3]

    mx = [list(x[i * 4 : (i + 1) * 4]) for i in range(4)]
    mt = [list(t2[i * 4 : (i + 1) * 4]) for i in range(4)]
    prod = [[0] * 4 for _ in range(4)]
    for row in range(4):
        for col in range(4):
            for k in range(4):
                prod[row][col] ^= gf_mul(mt[row][k], mx[k][col])
    y_rows = [prod[2], prod[3], prod[0], prod[1]]
    y = bytes(byte for row in y_rows for byte in row)

    z = bytearray(SBOX[x[i] ^ y[i] ^ t3[i]] for i in range(16))
    for i in range(1, 16):
        z[i] ^= z[i - 1]
    return x, y, bytes(z)


def invert_matrix_gf256(matrix: bytes) -> bytes:
    a = [[matrix[r * 4 + c] for c in range(4)] for r in range(4)]
    inv = [[1 if r == c else 0 for c in range(4)] for r in range(4)]

    def mul_inv(x: int) -> int:
        for y in range(256):
            if gf_mul(x, y) == 1:
                return y
        raise ValueError("matrix is singular")

    for col in range(4):
        pivot = next((row for row in range(col, 4) if a[row][col] != 0), None)
        if pivot is None:
            raise ValueError("matrix is singular")
        a[col], a[pivot] = a[pivot], a[col]
        inv[col], inv[pivot] = inv[pivot], inv[col]

        factor = mul_inv(a[col][col])
        for c in range(4):
            a[col][c] = gf_mul(a[col][c], factor)
            inv[col][c] = gf_mul(inv[col][c], factor)

        for row in range(4):
            if row == col:
                continue
            factor = a[row][col]
            if factor == 0:
                continue
            for c in range(4):
                a[row][c] ^= gf_mul(factor, a[col][c])
                inv[row][c] ^= gf_mul(factor, inv[col][c])

    return bytes(inv[r][c] for r in range(4) for c in range(4))


def apply_matrix_rowwise(matrix: bytes, data: bytes) -> bytes:
    out = bytearray(len(data))
    for row_idx in range(len(data) // 4):
        row = data[row_idx * 4 : (row_idx + 1) * 4]
        for col_idx in range(4):
            acc = 0
            for k in range(4):
                acc ^= gf_mul(matrix[col_idx * 4 + k], row[k])
            out[row_idx * 4 + col_idx] = acc
    return bytes(out)


def recover_flag(apk_path: str) -> str:
    with zipfile.ZipFile(apk_path, "r") as archive:
        cipher1 = archive.read("assets/cipher1.bin")
        cipher2 = archive.read("assets/cipher2.bin")
        cipher3 = archive.read("assets/cipher3.bin")
        puzzle = archive.read("assets/puzzle.bin")

    x, y, z = build_xyz()

    seg1_masked = tea_decrypt_block(cipher1, x)
    seg1 = bytes(a ^ b for a, b in zip(seg1_masked, z[1:9]))

    y_inv = invert_matrix_gf256(y)
    seg2 = apply_matrix_rowwise(y_inv, cipher2)

    seed = int.from_bytes(z[:8], "big")
    seg3 = bytes(a ^ b for a, b in zip(cipher3, lcg_stream(seed, len(cipher3))))

    pmat = puzzle[:16]
    pchk = puzzle[16:]
    for row in range(4):
        acc = 0
        for col in range(4):
            acc ^= gf_mul(pmat[row * 4 + col], seg1[col])
        if acc != pchk[row]:
            raise ValueError("puzzle constraint mismatch")

    xor_sum = reduce(lambda left, right: left ^ right, seg1, 0)
    expect_seg2_0 = CHARSET[(((xor_sum << 3) | (xor_sum >> 5)) & 0xFF) % len(CHARSET)]
    if seg2[0] != expect_seg2_0:
        raise ValueError("segment 2 constraint mismatch")

    mix_xor = 0
    mix_sum = 0
    for byte in seg1 + seg2:
        mix_xor ^= byte
        mix_sum = (mix_sum + byte) & 0xFF
    if seg3[8] != CHARSET[mix_xor % len(CHARSET)] or seg3[9] != CHARSET[mix_sum % len(CHARSET)]:
        raise ValueError("segment 3 tail constraint mismatch")

    if apply_matrix_rowwise(y, seg2) != cipher2:
        raise ValueError("segment 2 inversion failed")

    return (b"ISCC{" + seg1 + seg2 + seg3 + b"}").decode("ascii")


def main() -> int:
    if len(sys.argv) != 2:
        print(f"usage: python {sys.argv[0]} <attachment.apk>", file=sys.stderr)
        return 1

    try:
        print(recover_flag(sys.argv[1]))
        return 0
    except Exception as exc:
        print(f"error: {exc}", file=sys.stderr)
        return 2


if __name__ == "__main__":
    raise SystemExit(main())
```
