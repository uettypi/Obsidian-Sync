
## 详细解题分析

### Step 1 - 挖掘源码中的首关核心要素

在函数 `sg6j()` 内部，主要存在三种至关重要的数据：

1. `enc_key[]`：经过 XOR 混淆处理的移动密钥（move key）。
2. `MASK` 与 `EXPECT_OBF`：用于验证第一阶段哈希值的固定常量。
3. 源码注释部分存在的 `generated for stage1=...` 字段：该注释直接暴露了对应当前附件的 8 byte 第一阶段密码。

处理流程为：首先利用 `0x5A` 对 `enc_key[]` 执行异或解密，还原出真实的 move key；接着提取并保存注释里明文给出的 `pw1`。

### Step 2 - 模拟首关加密逻辑以计算阶段 Hash

分析源码可知，`g3::k1c7()` 函数主要执行了以下操作：

1. 针对输入明文进行区块大小为 8 bytes 的 PKCS#7 填充。
2. 依据解码出的 move 指令序列，对每一个 8 字节数据块执行先累加指定常量、再进行排列置换的操作。
3. 将上述处理完成的数据转化为 Base64 格式的字符串。

此后，程序会调用 `fnv1a32()` 对生成的 Base64 文本进行哈希计算，并核对是否满足以下条件：

```text
(stage1_hash ^ MASK) == EXPECT_OBF
```

基于此逻辑，解题脚本 `exp.py` 需要完全模仿 `k1c7()` 的变换过程以及 `fnv1a32()` 的哈希算法，代入前面获取到的 `pw1`，从而推算出 `stage1_hash`。

### Step 3 - 借助首关 Hash 衍生次关密码

通过审阅 `d4n7()` 函数可以发现，第二阶段并非一个孤立的挑战，其答案源自 `stage1_hash` 的进一步推演：

1. 首先利用之前得到的 hash 值去调整线性方程的系数 `g_a`。
2. 随后以 `seed = stage1_hash ^ 0x12345678` 作为随机数种子来初始化 LCG 算法。
3. 依次截取长度为 1 位、2 位、2 位、2 位、2 位的数值，组合形成五个必要的变量。

在此情境下，我们无需去强解那个线性方程组。只需要照搬源码内部的 LCG 生成机制，求得对应的 `x[0..4]` 序列，即可拼凑出当前环境下的第二段密码 `pw2`。

### Step 4 - 构建迷宫矩阵并搜索终局路线

程序内部通过如下公式：

```text
seed = int(pw2) ^ (stage1_hash * 0x9E3779B1)
```

初始化并生成一个规格为 4x4 的迷宫。迷宫的通路挖掘逻辑固定向右或向下延伸直达终点，并将途经的坐标单元格标识为通行状态。

由于目标源码中用于“产生额外岔路”的判定条件永远无法成立，因此该迷宫实际上退化为了一条单行道。尽管如此，为了保证解题脚本的兼容性，`exp.py` 依然采用了广度优先搜索（BFS）算法遍历此迷宫，找出由起点 `(0,0)` 前往终点 `(3,3)` 的最优连通路径，将其用作 `pw3`。

### Step 5 - 组装最终的目标 Flag

根据 `h5b5()` 函数的结构，最终结果的打印格式是写死的：

```text
ISCC{pw1pw2pw3}
```

综上所述，只要严格按照下发源码的实际内容计算出前置的三部分口令，便能成功构造出与之匹配的动态 Flag。

```python
#!/usr/bin/env python3
import argparse
import base64
import re
import ssl
import subprocess
import sys
import urllib.request
from collections import deque


DEFAULT_MAZE_WIDTH = 4
DEFAULT_MAZE_HEIGHT = 4
DEFAULT_LCG_MUL = 1664525
DEFAULT_LCG_ADD = 1013904223
DEFAULT_STAGE2_SEED_XOR = 0x12345678
DEFAULT_MAZE_HASH_MUL = 0x9E3779B1
DEFAULT_MAZE_SEED_XOR = 0x9E3779B9


def fnv1a32(data: bytes) -> int:
    h = 0x811C9DC5
    for b in data:
        h ^= b
        h = (h * 0x01000193) & 0xFFFFFFFF
    return h


def xor_decode(data: bytes, key: int) -> str:
    return bytes([b ^ key for b in data]).decode()


def parse_c_int(token: str) -> int:
    return int(token.rstrip("uUlL"), 0)


def parse_int_list(text: str) -> list[int]:
    return [int(x, 0) for x in re.findall(r"0x[0-9A-Fa-f]+|\d+", text)]


def parse_cpp_char_literal(token: str) -> int:
    body = token[1:-1]
    if body.startswith("\\"):
        decoded = bytes(body, "utf-8").decode("unicode_escape")
        if len(decoded) != 1:
            raise ValueError(f"unsupported cpp char literal: {token}")
        return ord(decoded)
    return ord(body)


def parse_enc_key(block_text: str) -> bytes:
    items = [item.strip() for item in block_text.split(",") if item.strip()]
    values = []
    for item in items:
        xor_match = re.search(r"\^\s*(0x[0-9A-Fa-f]+|\d+)", item)
        char_match = re.search(r"'((?:\\.|[^\\'])*)'", item)
        if xor_match and char_match:
            values.append(parse_cpp_char_literal("'" + char_match.group(1) + "'") ^ int(xor_match.group(1), 0))
            continue
        raw_values = parse_int_list(item)
        if len(raw_values) == 1:
            values.append(raw_values[0])
            continue
        raise ValueError(f"unsupported enc_key item: {item}")
    return bytes(values)


def fetch_source_text(path_or_url: str) -> str:
    try:
        with open(path_or_url, "r", encoding="utf-8", errors="ignore") as fh:
            return fh.read()
    except OSError:
        pass

    req = urllib.request.Request(path_or_url, headers={"User-Agent": "Mozilla/5.0", "Accept": "*/*"})
    ctx = ssl._create_unverified_context()
    try:
        with urllib.request.urlopen(req, timeout=60, context=ctx) as resp:
            return resp.read().decode("utf-8", errors="ignore")
    except Exception:
        pass

    proc = subprocess.run(["curl", "-L", "-k", "--max-time", "60", path_or_url], capture_output=True, text=True)
    if proc.returncode == 0 and proc.stdout:
        return proc.stdout
    raise RuntimeError(f"failed to read source from {path_or_url}")


def extract_named_constant(text: str, name_or_value: str) -> int | None:
    if re.fullmatch(r"0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*", name_or_value):
        return parse_c_int(name_or_value)
    match = re.search(
        rf"static\s+const\s+uint32_t\s+{re.escape(name_or_value)}\s*=\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)",
        text,
    )
    if match:
        return parse_c_int(match.group(1))
    return None


def extract_define_int(text: str, name: str, default: int) -> int:
    match = re.search(rf"#define\s+{re.escape(name)}\s+(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)", text)
    return parse_c_int(match.group(1)) if match else default


def extract_source_params(source_text: str):
    key_call_match = re.search(
        r"std::string\s+key\s*=\s*h5m77\(\s*([A-Za-z_]\w*)\s*,\s*sizeof\(\1\)\s*,\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\)",
        source_text,
    )
    if not key_call_match:
        raise RuntimeError("failed to locate decoded stage1 key call")

    enc_array_name = key_call_match.group(1)
    key_xor = parse_c_int(key_call_match.group(2))
    enc_key_match = re.search(
        rf"static\s+const\s+uint8_t\s+{re.escape(enc_array_name)}\[\]\s*=\s*\{{(.*?)\}};",
        source_text,
        re.S,
    )
    if not enc_key_match:
        raise RuntimeError(f"failed to locate stage1 key array {enc_array_name}")

    stage1_region_match = re.search(
        r"g_stage1_hash\s*=\s*h\s*;(.*?return\s*\(\(\s*h\s*\^\s*([A-Za-z_]\w*|0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\)\s*==\s*([A-Za-z_]\w*|0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\)\s*;)",
        source_text,
        re.S,
    )
    if not stage1_region_match:
        raise RuntimeError("failed to locate stage1 compare expression")

    stage1_region = stage1_region_match.group(1)
    mask = extract_named_constant(source_text, stage1_region_match.group(2))
    expect_obf = extract_named_constant(source_text, stage1_region_match.group(3))
    if mask is None or expect_obf is None:
        raise RuntimeError("failed to resolve stage1 compare constants")

    stage1_match = re.search(r"stage1\s*=\s*([!-~]{8})", stage1_region)
    if not stage1_match:
        stage1_match = re.search(r"stage1\s*=\s*([!-~]{8})", source_text)
    if not stage1_match:
        raise RuntimeError("failed to locate generated stage1 plaintext hint")

    lcg_match = re.search(
        r"\*st\s*=\s*\(\*st\s*\*\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\+\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\)",
        source_text,
    )
    lcg_mul = parse_c_int(lcg_match.group(1)) if lcg_match else DEFAULT_LCG_MUL
    lcg_add = parse_c_int(lcg_match.group(2)) if lcg_match else DEFAULT_LCG_ADD

    stage2_seed_match = re.search(r"seed\s*=\s*h\s*\^\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)", source_text)
    stage2_seed_xor = parse_c_int(stage2_seed_match.group(1)) if stage2_seed_match else DEFAULT_STAGE2_SEED_XOR

    maze_hash_mul_match = re.search(
        r"seed\s*\^=\s*\(\s*g_stage1_hash\s*\*\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)\s*\)",
        source_text,
    )
    maze_hash_mul = parse_c_int(maze_hash_mul_match.group(1)) if maze_hash_mul_match else DEFAULT_MAZE_HASH_MUL

    maze_seed_xor_match = re.search(r"st\s*=\s*seed\s*\^\s*(0x[0-9A-Fa-f]+[uUlL]*|\d+[uUlL]*)", source_text)
    maze_seed_xor = parse_c_int(maze_seed_xor_match.group(1)) if maze_seed_xor_match else DEFAULT_MAZE_SEED_XOR

    maze_width = extract_define_int(source_text, "MAZE_WIDTH", DEFAULT_MAZE_WIDTH)
    maze_height = extract_define_int(source_text, "MAZE_HEIGHT", DEFAULT_MAZE_HEIGHT)

    return {
        "enc_key": parse_enc_key(enc_key_match.group(1)),
        "key_xor": key_xor,
        "mask": mask,
        "expect_obf": expect_obf,
        "pw1": stage1_match.group(1),
        "lcg_mul": lcg_mul,
        "lcg_add": lcg_add,
        "stage2_seed_xor": stage2_seed_xor,
        "maze_hash_mul": maze_hash_mul,
        "maze_seed_xor": maze_seed_xor,
        "maze_width": maze_width,
        "maze_height": maze_height,
    }


def lcg_step(state: int, mul: int, add: int) -> int:
    return (state * mul + add) & 0xFFFFFFFF


def pkcs7_pad(data: bytes, block: int = 8) -> bytes:
    pad = block - (len(data) % block)
    if pad == 0:
        pad = block
    return data + bytes([pad]) * pad


def invert_perm(perm):
    inv = [0] * 8
    for old, new in enumerate(perm):
        inv[new] = old
    return inv


def make_add(indices, value):
    arr = [0] * 8
    for idx in indices:
        arr[idx] = value
    return arr


def build_moves():
    moves = {
        "R": {"perm": [0, 5, 2, 1, 4, 7, 6, 3]},
        "L": {"perm": [2, 1, 6, 3, 0, 5, 4, 7]},
        "U": {"perm": [0, 1, 6, 2, 4, 5, 7, 3]},
        "D": {"perm": [1, 5, 2, 3, 0, 4, 6, 7]},
        "F": {"perm": [0, 1, 2, 3, 6, 4, 7, 5]},
        "B": {"perm": [1, 3, 0, 2, 4, 5, 6, 7]},
    }
    moves["R"]["add"] = make_add([1, 3, 5, 7], 1)
    moves["L"]["add"] = make_add([0, 2, 4, 6], 2)
    moves["U"]["add"] = make_add([2, 3, 6, 7], 3)
    moves["D"]["add"] = make_add([0, 1, 4, 5], 4)
    moves["F"]["add"] = make_add([4, 5, 6, 7], 5)
    moves["B"]["add"] = make_add([0, 1, 2, 3], 6)
    moves["R'"] = {"perm": invert_perm(moves["R"]["perm"]), "add": make_add([1, 3, 5, 7], 7)}
    moves["L'"] = {"perm": invert_perm(moves["L"]["perm"]), "add": make_add([0, 2, 4, 6], 8)}
    moves["U'"] = {"perm": invert_perm(moves["U"]["perm"]), "add": make_add([2, 3, 6, 7], 9)}
    moves["D'"] = {"perm": invert_perm(moves["D"]["perm"]), "add": make_add([0, 1, 4, 5], 10)}
    moves["F'"] = {"perm": invert_perm(moves["F"]["perm"]), "add": make_add([4, 5, 6, 7], 11)}
    moves["B'"] = {"perm": invert_perm(moves["B"]["perm"]), "add": make_add([0, 1, 2, 3], 12)}
    return moves


def expand_key(key: str, moves):
    seq = []
    for tok in key.split():
        tok = tok.upper()
        times = 1
        if tok.endswith("2"):
            times = 2
            tok = tok[:-1]
        seq.extend([tok] * times)
    return seq


def permute_block(block, perm):
    out = [0] * 8
    for old in range(8):
        out[perm[old]] = block[old]
    return out


def apply_move(block, spec):
    tmp = [(block[i] + spec["add"][i]) & 0xFF for i in range(8)]
    return permute_block(tmp, spec["perm"])


def stage1_cipher(plaintext: str, key: str) -> str:
    moves = build_moves()
    seq = expand_key(key, moves)
    data = pkcs7_pad(plaintext.encode(), 8)
    out = []
    for off in range(0, len(data), 8):
        block = list(data[off:off + 8])
        for name in seq:
            block = apply_move(block, moves[name])
        out.extend(block)
    return base64.b64encode(bytes(out)).decode()


def derive_stage2_vector(stage1_hash: int, seed_xor: int, lcg_mul: int, lcg_add: int) -> list[int]:
    state = stage1_hash ^ seed_xor
    out = []
    for i in range(5):
        state = lcg_step(state, lcg_mul, lcg_add)
        out.append(state % 10 if i == 0 else 10 + (state % 90))
    return out


def build_stage2_password(x: list[int]) -> str:
    return f"{x[0]}{x[1]:02d}{x[2]:02d}{x[3]:02d}{x[4]:02d}"


def build_maze(seed: int, width: int, height: int, maze_seed_xor: int, lcg_mul: int, lcg_add: int):
    maze = [0] * (width * height)
    state = seed ^ maze_seed_xor
    x = y = 0
    maze[0] = 1
    goal_x = width - 1
    goal_y = height - 1
    while not (x == goal_x and y == goal_y):
        state = lcg_step(state, lcg_mul, lcg_add)
        if x < goal_x and y < goal_y:
            move = 1 if (state & 1) else 2
        elif x < goal_x:
            move = 1
        else:
            move = 2
        if move == 1:
            x += 1
        else:
            y += 1
        maze[y * width + x] = 1
    return maze


def find_shortest_maze_path(maze, width: int, height: int):
    queue = deque([(0, 0, "")])
    seen = {(0, 0)}
    while queue:
        x, y, path = queue.popleft()
        if (x, y) == (width - 1, height - 1):
            return path
        for ch, dx, dy in (("L", -1, 0), ("R", 1, 0), ("U", 0, -1), ("D", 0, 1)):
            nx = x + dx
            ny = y + dy
            if not (0 <= nx < width and 0 <= ny < height):
                continue
            if maze[ny * width + nx] == 0:
                continue
            if (nx, ny) in seen:
                continue
            seen.add((nx, ny))
            queue.append((nx, ny, path + ch))
    raise RuntimeError("maze has no valid path")


def solve(path_or_url: str) -> str:
    source_text = fetch_source_text(path_or_url)
    params = extract_source_params(source_text)
    key = xor_decode(params["enc_key"], params["key_xor"])
    pw1 = params["pw1"]
    cipher = stage1_cipher(pw1, key)
    stage1_hash = fnv1a32(cipher.encode())
    if (stage1_hash ^ params["mask"]) != params["expect_obf"]:
        raise RuntimeError("stage1 verification mismatch")
    stage2_vector = derive_stage2_vector(
        stage1_hash,
        params["stage2_seed_xor"],
        params["lcg_mul"],
        params["lcg_add"],
    )
    pw2 = build_stage2_password(stage2_vector)
    maze_seed = int(pw2) ^ ((stage1_hash * params["maze_hash_mul"]) & 0xFFFFFFFF)
    maze = build_maze(
        maze_seed,
        params["maze_width"],
        params["maze_height"],
        params["maze_seed_xor"],
        params["lcg_mul"],
        params["lcg_add"],
    )
    pw3 = find_shortest_maze_path(maze, params["maze_width"], params["maze_height"])
    return f"ISCC{{{pw1}{pw2}{pw3}}}"


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("source")
    args = parser.parse_args()
    print(solve(args.source))
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
