
## 解题过程

### 步骤 1 - 初始侦察与脱壳

**关键观察**

附件内只有一个 Windows x64 PE，可见 UPX0/UPX1/UPX2 段，说明第一步是标准 UPX 脱壳。脱壳前没有明文 flag，但已经能看到格式提示与输入长度检查相关字符串。

**关键命令**

```bash
unzip -o attachment-20.zip -d work
file work/caterpillar7.exe
rabin2 -S work/caterpillar7.exe
upx -d work/caterpillar7.exe -o work/caterpillar7.unpacked.exe
strings -a -n 4 work/caterpillar7.unpacked.exe | rg 'ISCC|Invalid|Success|Fail'
```

**关键输出**


### 步骤 2 - 定位主逻辑并确认双分支成功条件

**关键观察**

反编译 main 后可以看到：先验证总长度为 26、前缀为 ISCC{、后缀为 }，再取出 20 字节主体。随后根据两次取时间的差值决定走哪条分支：时间差大于 5.0 进入慢分支，比较两份 SHA-256 十六进制串；否则进入快分支，对主体做 LCG 驱动的可打印字符环移。最后结果由 bVar4 || bVar5 汇总，因此两条分支是 OR，不是 AND。

**关键命令**

```bash
rizin -A -q -c 's main; pdg' work/caterpillar7.unpacked.exe
```

**关键输出**

### 步骤 3 - 提取 .data 参数并逆向快分支

**关键观察**

快分支真正依赖的参数都落在 .data 头部：偏移 +0x10 是 seed，偏移 +0x18 是 20 字符 target。当前样本中 seed=7，target=WY|lFW#ImVR&OKNX-%Ji。变换是标准可打印字符环上的加法，因此直接逐字节减去同一 LCG 产生的位移即可恢复原始主体。

**关键命令**

```bash
objdump -s -j .data work/caterpillar7.unpacked.exe | sed -n '1,8p'
python3 - <<'PY'
target='WY|lFW#ImVR&OKNX-%Ji'
state=7
out=[]
for ch in target:
    state=(state*0x41c64e6d + 0x3039) & 0xffffffff
    shift=(state>>24)%95
    out.append(chr(((ord(ch)-0x20-shift)%95)+0x20))
print(''.join(out))
PY
```

**关键输出**

```text

14003f020 6d565226 4f4b4e58 2d254a69 00000000
qUV|-(Ny[O%mu5kLIBzJ
```

### 步骤 4 - 通用化 EXP 并验证动态附件场景

**关键观察**

为了适配同构动态附件，EXP 不写死 seed、target 或 flag。脚本会自动处理 zip、自动脱壳 UPX、解析 PE .data，取出 seed 与 target 后逆向快分支。只要题目仍然沿用这套布局与算法，替换附件也能直接恢复对应 flag。

```python
#!/usr/bin/env python3
import argparse
import os
import shutil
import struct
import subprocess
import sys
import tempfile
import zipfile
from pathlib import Path


def read_c_string(buf: bytes, start: int) -> bytes:
    end = buf.find(b"\x00", start)
    if end == -1:
        raise ValueError(f"unterminated c-string at offset 0x{start:x}")
    return buf[start:end]


def unpack_if_needed(src: Path) -> Path:
    data = src.read_bytes()
    if b"UPX0" not in data or b"UPX1" not in data:
        return src

    upx = shutil.which("upx") or shutil.which("upx-ucl")
    if not upx:
        return src

    fd, tmp_name = tempfile.mkstemp(prefix=src.stem + ".", suffix=".unpacked.exe")
    os.close(fd)
    tmp_path = Path(tmp_name)
    shutil.copy2(src, tmp_path)
    try:
        subprocess.run([upx, "-d", str(tmp_path)], check=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        return tmp_path
    except subprocess.CalledProcessError:
        tmp_path.unlink(missing_ok=True)
        return src


def resolve_binary_path(src: Path):
    if src.is_dir():
        exes = sorted(src.rglob("*.exe"))
        if len(exes) != 1:
            raise ValueError(f"expected exactly one exe in directory, found {len(exes)}")
        return exes[0], None

    if src.suffix.lower() == ".zip":
        td = tempfile.TemporaryDirectory(prefix=src.stem + ".")
        with zipfile.ZipFile(src) as zf:
            zf.extractall(td.name)
        exes = sorted(Path(td.name).rglob("*.exe"))
        if len(exes) != 1:
            td.cleanup()
            raise ValueError(f"expected exactly one exe in zip, found {len(exes)}")
        return exes[0], td

    return src, None


class PEView:
    def __init__(self, path: Path):
        self.path = path
        self.data = path.read_bytes()
        self.image_base = 0
        self.sections = []
        self._parse()

    def _parse(self) -> None:
        if self.data[:2] != b"MZ":
            raise ValueError("not a PE file")
        e_lfanew = struct.unpack_from("<I", self.data, 0x3C)[0]
        if self.data[e_lfanew:e_lfanew + 4] != b"PE\x00\x00":
            raise ValueError("invalid PE signature")

        coff = e_lfanew + 4
        number_of_sections = struct.unpack_from("<H", self.data, coff + 2)[0]
        size_of_optional_header = struct.unpack_from("<H", self.data, coff + 16)[0]
        optional = coff + 20
        magic = struct.unpack_from("<H", self.data, optional)[0]
        if magic != 0x20B:
            raise ValueError("expected PE32+")

        self.image_base = struct.unpack_from("<Q", self.data, optional + 24)[0]
        section_table = optional + size_of_optional_header

        for idx in range(number_of_sections):
            off = section_table + idx * 40
            name = self.data[off:off + 8].split(b"\x00", 1)[0].decode("ascii", "ignore")
            virtual_size, virtual_address, raw_size, raw_ptr = struct.unpack_from("<IIII", self.data, off + 8)
            self.sections.append(
                {
                    "name": name,
                    "vsize": virtual_size,
                    "vaddr": virtual_address,
                    "raw_size": raw_size,
                    "raw_ptr": raw_ptr,
                }
            )

    def section(self, name: str):
        for sec in self.sections:
            if sec["name"] == name:
                return sec
        raise KeyError(f"section {name} not found")

    def rva_to_offset(self, rva: int) -> int:
        for sec in self.sections:
            start = sec["vaddr"]
            size = max(sec["vsize"], sec["raw_size"])
            end = start + size
            if start <= rva < end:
                return sec["raw_ptr"] + (rva - start)
        raise KeyError(f"rva 0x{rva:x} not in any section")

    def va_to_offset(self, va: int) -> int:
        return self.rva_to_offset(va - self.image_base)


def extract_parameters(pe: PEView):
    data_sec = pe.section(".data")
    start = data_sec["raw_ptr"]
    blob = pe.data[start:start + data_sec["raw_size"]]

    slow_hash_1_ptr = struct.unpack_from("<Q", blob, 0x00)[0]
    slow_hash_2_ptr = struct.unpack_from("<Q", blob, 0x08)[0]
    seed = struct.unpack_from("<I", blob, 0x10)[0]
    target = read_c_string(blob, 0x18).decode("ascii")

    slow_hash_1 = read_c_string(pe.data, pe.va_to_offset(slow_hash_1_ptr)).decode("ascii")
    slow_hash_2 = read_c_string(pe.data, pe.va_to_offset(slow_hash_2_ptr)).decode("ascii")

    return {
        "seed": seed,
        "target": target,
        "hashes": [slow_hash_1, slow_hash_2],
    }


def invert_fast_branch(target: str, seed: int) -> str:
    state = seed & 0xFFFFFFFF
    out = []
    for ch in target:
        state = (state * 0x41C64E6D + 0x3039) & 0xFFFFFFFF
        shift = (state >> 24) % 95
        plain = ((ord(ch) - 0x20 - shift) % 95) + 0x20
        out.append(chr(plain))
    return "".join(out)


def main() -> int:
    parser = argparse.ArgumentParser(description="Recover the fast-branch flag from caterpillar-style dynamic attachments.")
    parser.add_argument("binary", help="path to the challenge exe")
    parser.add_argument("--prefix", default="ISCC{", help="flag prefix")
    parser.add_argument("--suffix", default="}", help="flag suffix")
    args = parser.parse_args()

    binary = Path(args.binary).resolve()
    if not binary.exists():
        print(f"[-] file not found: {binary}", file=sys.stderr)
        return 1

    resolved, extracted = resolve_binary_path(binary)
    unpacked = unpack_if_needed(resolved)
    cleanup = unpacked != resolved
    try:
        pe = PEView(unpacked)
        params = extract_parameters(pe)
        body = invert_fast_branch(params["target"], params["seed"])
        flag = f"{args.prefix}{body}{args.suffix}"

        print(f"[+] binary      : {binary}")
        if resolved != binary:
            print(f"[+] extracted   : {resolved}")
        if cleanup:
            print(f"[+] unpacked    : {unpacked}")
        print(f"[+] seed        : {params['seed']}")
        print(f"[+] target      : {params['target']}")
        print(f"[+] fast flag   : {flag}")
        print(f"[+] slow hashes : {params['hashes'][0]}")
        print(f"                {params['hashes'][1]}")
        print("[*] note        : this recovers the <=5s branch by reversing the printable LCG shift.")
        return 0
    finally:
        if cleanup:
            unpacked.unlink(missing_ok=True)
        if extracted is not None:
            extracted.cleanup()


if __name__ == "__main__":
    raise SystemExit(main())
```
